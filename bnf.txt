Program ::=
    Assign 	| assign variable := expression ; program
    Print 	| print expression
    For 	| for term times program ; program
    If 		| if formula program ; program
    End 	| end

Expression ::= term | formula

Formula ::=
    And 	| formula and formula
    Or 		| formula or formula
    Not 	| not formula
    GT 		| term gt term
    LT 		| term lt  term
    GTE 	| term gte term
    LTE 	| term lte term
    Equal   | term eq term
    True    | true
    False   | false

Term ::= 
    Plus    | term + term
    Minus   | term - term
    Factor  | factor

Factor ::= number | variable
    Mult 	| term * term
    Div 	| term / term
    Power 	| term ^ term
    Mod 	| term % term
    Number  | (0[1-9][0-9]*)
    Variable| [a-z][A-Za-z]*
